var posX;
var vitX;
function setup(){
	createCanvas(400,100);
	fill(255,0,0);
	noStroke();
	posX=20;
	vitX=2;
}
function draw(){
	background(240);
	if (posX<20){
		vitX=2;
	}
	if (posX>380){
		vitX=-2;
	}
	posX=posX+vitX;
	pacMan(vitX);
}
function pacMan(s){
	var d;
	var angle_dep;
	var angle_arr;
	if (s>0){
		d=8;
		angle_dep=radians(30);
		angle_arr=radians(330);
	}
	else{
		d=-8;
		angle_dep=radians(-150);
		angle_arr=radians(150);
	}
	fill(255,230,0);
	arc(posX,50,40,40,angle_dep,angle_arr);
	fill(0);
	ellipse(posX+d,33,5,5);
}

Proposition solution "À faire vous-même 6.7"


var posX1;
var posY1;
var vitX1;
var vitY1;
var posX2;
var posY2;
var vitX2;
var vitY2;
var collision=false;
function setup(){
	createCanvas(200,200);
	fill(255,0,0);
	noStroke();
	posX1=random(10,190);
	posY1=random(10,190);
	vitX1=random(-3,3);
	vitY1=random(-3,3);
	posX2=random(10,190);
	posY2=random(10,190);
	vitX2=random(-3,3);
	vitY2=random(-3,3);
}
function draw(){
	background(240);
	if (collision==false){
		deplacement();
		contact();
		fill(255,0,0);
		ellipse(posX1,posY1,20,20);
		fill(0,255,0);
		ellipse(posX2,posY2,20,20);
	}
}
function deplacement(){
	if (posX1<10 || posX1>190){
		vitX1=-vitX1;
	}
	if (posY1<10 || posY1>190){
		vitY1=-vitY1;
	}
	if (posX2<10 || posX2>190){
		vitX2=-vitX2;
	}
	if (posY2<10 || posY2>190){
		vitY2=-vitY2;
	}
	posX1=posX1+vitX1;
	posY1=posY1+vitY1;
	posX2=posX2+vitX2;
	posY2=posY2+vitY2;
}
function contact(){
	var d=sqrt(pow((posX1-posX2),2)+pow((posY1-posY2),2));
	if (d<20){
		collision=true;
	}
}

Proposition solution "À faire vous-même 8.3"


var key_d;
var key_g;
var key_h;
var key_b;
var x;
var y;
function setup(){
	createCanvas(400,400);
	stroke(0,0,0);
	fill(0,0,0);
	key_d=false;
	key_g=false;
	key_h=false;
	key_b=false;
	x=200;
	y=200;
}
function draw(){
	background(240);
	deplacement();
	ellipse(x,y,20,20)
}
function keyPressed(){
	if (keyCode==39){
		key_d=true;
	}
	if (keyCode==37){
		key_g=true;
	}
	if (keyCode==38){
		key_h=true;
	}
	if (keyCode==40){
		key_b=true;
	}
}
function keyReleased(){
	if (keyCode==39){
		key_d=false;
	}
	if (keyCode==37){
		key_g=false;
	}
	if (keyCode==38){
		key_h=false;
	}
	if (keyCode==40){
		key_b=false;
	}
}
function deplacement(){
	if (key_d==true){
		x=x+2;
	}
	if (key_g==true){
		x=x-2;
	}
	if (key_h==true){
		y=y-2;
	}
	if (key_b==true){
		y=y+2;
	}
}
